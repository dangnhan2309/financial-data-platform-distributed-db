SQLPLUS / AS SYSDBA;


ALTER SESSION SET CONTAINER = FARM_SITE_BAC;
ALTER SESSION SET CURRENT_SCHEMA = FARM_ADMIN;

BEGIN
  FOR t IN (
    SELECT table_name
    FROM all_tables
    WHERE owner = 'FARM_ADMIN'
      AND table_name IN (
        'FARM','WAREHOUSE','SUPPLIER','INGREDIENT','PRODUCT','FARM_PLOT','FARM_BED','IOT_SENSOR','SENSOR_DATA','CULTIVATION_LOG',
        'PURCHASE_CONTRACT','PURCHASE_BATCH','PURCHASE_TICKET','WEIGHING_TICKET','PRODUCTION_LOT','QC_RECORD','TRANSPORT_TICKET','INVENTORY'
      )
  ) LOOP
    BEGIN
      EXECUTE IMMEDIATE 'DROP MATERIALIZED VIEW LOG ON ' || t.table_name;
    EXCEPTION WHEN OTHERS THEN NULL; END;
    EXECUTE IMMEDIATE 'CREATE MATERIALIZED VIEW LOG ON ' || t.table_name || ' WITH PRIMARY KEY, ROWID INCLUDING NEW VALUES';
  END LOOP;
END;
/


ALTER SESSION SET CONTAINER = FARM_SITE;
GRANT CREATE DATABASE LINK TO farm_admin;


CONNECT farm_admin/Aa123@localhost:1521/FARM_SITE;

  USING '(DESCRIPTION=(ADDRESS=(PROTOCOL=TCP)(HOST=localhost)(PORT=1521))(CONNECT_DATA=(SERVICE_NAME=FARM_SITE_BAC)))';
