CREATE PLUGGABLE DATABASE FARM_SITE ADMIN USER farm_admin IDENTIFIED BY "Aa123"
	FILE_NAME_CONVERT = ('pdbseed', 'farm_site_bac');

-- 1) Create PDB cho 3 miền (BAC/TRUNG/NAM)
-- 
-- Nếu PDB đã tồn tại, cần CLOSE IMMEDIATE trước khi DROP
ALTER PLUGGABLE DATABASE FARM_SITE_BAC CLOSE IMMEDIATE;
DROP PLUGGABLE DATABASE FARM_SITE_BAC INCLUDING DATAFILES;
CREATE PLUGGABLE DATABASE FARM_SITE_BAC ADMIN USER farm_admin IDENTIFIED BY "Aa123"
	FILE_NAME_CONVERT = ('pdbseed', 'farm_site_bac');
-- 
ALTER PLUGGABLE DATABASE FARM_SITE_TRUNG CLOSE IMMEDIATE;
DROP PLUGGABLE DATABASE FARM_SITE_TRUNG INCLUDING DATAFILES; 
CREATE PLUGGABLE DATABASE FARM_SITE_TRUNG ADMIN USER farm_admin IDENTIFIED BY "Aa123"
	FILE_NAME_CONVERT = ('pdbseed', 'farm_site_trung');
-- 
ALTER PLUGGABLE DATABASE FARM_SITE_NAM CLOSE IMMEDIATE;
DROP PLUGGABLE DATABASE FARM_SITE_NAM INCLUDING DATAFILES; 
CREATE PLUGGABLE DATABASE FARM_SITE_NAM ADMIN USER farm_admin IDENTIFIED BY "Aa123"
	FILE_NAME_CONVERT = ('pdbseed', 'farm_site_nam');

-- 2) Mở PDB và lưu state
ALTER PLUGGABLE DATABASE FARM_SITE_BAC OPEN;
ALTER PLUGGABLE DATABASE FARM_SITE_TRUNG OPEN;
ALTER PLUGGABLE DATABASE FARM_SITE_NAM OPEN;

ALTER PLUGGABLE DATABASE FARM_SITE_BAC SAVE STATE;
ALTER PLUGGABLE DATABASE FARM_SITE_TRUNG SAVE STATE;
ALTER PLUGGABLE DATABASE FARM_SITE_NAM SAVE STATE;

-- 3) Tạo user local farm_admin cho từng PDB (cách ly theo vùng; không cấp SET CONTAINER)

-- Miền BAC
ALTER SESSION SET CONTAINER = FARM_SITE_BAC;
BEGIN
	EXECUTE IMMEDIATE 'CREATE TABLESPACE USERS DATAFILE ''farm_site_bac_users01.dbf'' SIZE 200M AUTOEXTEND ON NEXT 50M';
EXCEPTION WHEN OTHERS THEN NULL; END;
/
DECLARE
	cnt NUMBER;
BEGIN
	SELECT COUNT(*) INTO cnt FROM dba_users WHERE username = 'FARM_ADMIN';
	IF cnt = 0 THEN
		EXECUTE IMMEDIATE 'CREATE USER farm_admin IDENTIFIED BY "Aa123" DEFAULT TABLESPACE USERS QUOTA UNLIMITED ON USERS';
	ELSE
		EXECUTE IMMEDIATE 'ALTER USER farm_admin IDENTIFIED BY "Aa123" QUOTA UNLIMITED ON USERS';
	END IF;
END;
/
GRANT CONNECT, RESOURCE TO farm_admin;
