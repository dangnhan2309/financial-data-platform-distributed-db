CREATE TABLE ingredient (
    ingredient_id VARCHAR2(50) PRIMARY KEY,
    name VARCHAR2(200),
    unit_of_measure VARCHAR2(50),
    is_active NUMBER(1)
);

CREATE TABLE packaging_specification (
    packaging_id VARCHAR2(50) PRIMARY KEY,
    name VARCHAR2(200),
    unit_of_measure VARCHAR2(50),
    volume NUMBER,
    material VARCHAR2(100)
);

CREATE TABLE certification (
    certification_id VARCHAR2(50) PRIMARY KEY,
    name VARCHAR2(200),
    issue_date DATE,
    expiry_date DATE,
    issued_by VARCHAR2(200)
);

CREATE TABLE quality_assurance (
    quality_assurance_id VARCHAR2(50) PRIMARY KEY,
    name VARCHAR2(200),
    description VARCHAR2(500)
);

CREATE TABLE product_type (
    product_type_id VARCHAR2(50) PRIMARY KEY,
    name VARCHAR2(200)
);

CREATE TABLE factory (
    factory_id VARCHAR2(50) PRIMARY KEY,
    name VARCHAR2(200),
    address VARCHAR2(500)
);

CREATE TABLE incoterm (
    incoterm_id VARCHAR2(50) PRIMARY KEY,
    name VARCHAR2(100),
    description VARCHAR2(500)
);

CREATE TABLE customer_type (
    customer_type_id VARCHAR2(50) PRIMARY KEY,
    name VARCHAR2(100)
);

CREATE TABLE contract_type (
    contract_type_id VARCHAR2(50) PRIMARY KEY,
    name VARCHAR2(100)
);

CREATE TABLE payment_term (
    payment_term_id VARCHAR2(50) PRIMARY KEY,
    description VARCHAR2(500)
);

CREATE TABLE staff (
    staff_id VARCHAR2(50) PRIMARY KEY,
    name VARCHAR2(200),
    role VARCHAR2(100),
    email VARCHAR2(200),
    phone VARCHAR2(50)
);

CREATE TABLE bank (
    bank_id VARCHAR2(50) PRIMARY KEY,
    bank_name VARCHAR2(200),
    account_name VARCHAR2(200),
    account_number VARCHAR2(100),
    swift_code VARCHAR2(50),
    bank_address VARCHAR2(500),
    country VARCHAR2(100)
);
CREATE TABLE product (
    product_id VARCHAR2(50) PRIMARY KEY,
    product_type_id VARCHAR2(50),
    name VARCHAR2(200),
    description VARCHAR2(500),
    price NUMBER,
    application VARCHAR2(200),
    brix VARCHAR2(50),
    size VARCHAR2(50),
    solid NUMBER,
    ph NUMBER,
    is_active NUMBER(1),

    CONSTRAINT fk_product_type
    FOREIGN KEY (product_type_id)
    REFERENCES product_type(product_type_id)
);
CREATE TABLE ingredient_product_detail (
    ingredient_id VARCHAR2(50),
    product_id VARCHAR2(50),
    quantity NUMBER,
    unit_of_measure VARCHAR2(50),
    CONSTRAINT pk_ing_prod
    PRIMARY KEY (ingredient_id, product_id),
    CONSTRAINT fk_ing
    FOREIGN KEY (ingredient_id)
    REFERENCES ingredient(ingredient_id),
    CONSTRAINT fk_prod
    FOREIGN KEY (product_id)
    REFERENCES product(product_id)
);
CREATE TABLE product_specification (
    product_specification_id VARCHAR2(50) PRIMARY KEY,
    product_id VARCHAR2(50),
    packaging_id VARCHAR2(50),
    certification_id VARCHAR2(50),
    quality_assurance_id VARCHAR2(50),
    shelf_life_days NUMBER,
    storage_condition VARCHAR2(200),
    origin VARCHAR2(200),
    FOREIGN KEY (product_id)
    REFERENCES product(product_id),
    FOREIGN KEY (packaging_id)
    REFERENCES packaging_specification(packaging_id),
    FOREIGN KEY (certification_id)
    REFERENCES certification(certification_id),
    FOREIGN KEY (quality_assurance_id)
    REFERENCES quality_assurance(quality_assurance_id)
);

CREATE TABLE production_lot (
    production_lot_id VARCHAR2(50) PRIMARY KEY,
    factory_id VARCHAR2(50),
    product_id VARCHAR2(50),
    production_date DATE,
    expiry_date DATE,
    status NUMBER(1),
    FOREIGN KEY (factory_id)
    REFERENCES factory(factory_id),
    FOREIGN KEY (product_id)
    REFERENCES product(product_id)
);
CREATE TABLE certificate_of_analysis (
    certificate_of_analysis_id VARCHAR2(50) PRIMARY KEY,
    lot_id VARCHAR2(50),
    issue_date DATE,
    FOREIGN KEY (lot_id)
    REFERENCES production_lot(production_lot_id)
);
CREATE TABLE customer (
    customer_id VARCHAR2(50) PRIMARY KEY,
    customer_type_id VARCHAR2(50),
    customer_code VARCHAR2(50),
    company_name VARCHAR2(200),
    short_name VARCHAR2(100),
    tax_id VARCHAR2(100),
    country VARCHAR2(100),
    city VARCHAR2(100),
    address VARCHAR2(300),
    phone VARCHAR2(50),
    email VARCHAR2(200),
    website VARCHAR2(200),
    industry VARCHAR2(200),
    status VARCHAR2(50),
    preferred_currency VARCHAR2(50),
    FOREIGN KEY (customer_type_id)
    REFERENCES customer_type(customer_type_id)
);
CREATE TABLE quotation (
    quotation_id VARCHAR2(50) PRIMARY KEY,
    customer_id VARCHAR2(50),
    staff_id VARCHAR2(50),
    quotation_date DATE,
    expiry_date DATE,
    incoterm_id VARCHAR2(50),
    payment_term_id VARCHAR2(50),
    FOREIGN KEY (customer_id)
    REFERENCES customer(customer_id),
    FOREIGN KEY (staff_id)
    REFERENCES staff(staff_id),
    FOREIGN KEY (incoterm_id)
    REFERENCES incoterm(incoterm_id),
    FOREIGN KEY (payment_term_id)
    REFERENCES payment_term(payment_term_id)
);
CREATE TABLE quotation_item (
    quotation_id VARCHAR2(50),
    product_id VARCHAR2(50),
    quantity NUMBER,
    unit_price NUMBER,
    CONSTRAINT pk_quotation_item
    PRIMARY KEY (quotation_id, product_id),
    FOREIGN KEY (quotation_id)
    REFERENCES quotation(quotation_id),
    FOREIGN KEY (product_id)
    REFERENCES product(product_id)
);
CREATE TABLE proforma_invoice (
    proforma_invoice_id VARCHAR2(50) PRIMARY KEY,
    quotation_id VARCHAR2(50),
    payment_term_id VARCHAR2(50),
    total_contract_value NUMBER,
    bank_id VARCHAR2(50),
    staff_id VARCHAR2(50),
    port_of_loading VARCHAR2(100),
    port_of_discharge VARCHAR2(100),
    delivery_time DATE,
    file_path VARCHAR2(500),
    FOREIGN KEY (quotation_id)
    REFERENCES quotation(quotation_id),
    FOREIGN KEY (payment_term_id)
    REFERENCES payment_term(payment_term_id),
    FOREIGN KEY (bank_id)
    REFERENCES bank(bank_id),
    FOREIGN KEY (staff_id)
    REFERENCES staff(staff_id)
);


CREATE TABLE contract (
    contract_id VARCHAR2(50) PRIMARY KEY,
    customer_id VARCHAR2(50),
    payment_term_id VARCHAR2(50),
    proforma_invoice_id VARCHAR2(50),
    staff_id VARCHAR2(50),
    contract_type_id VARCHAR2(50),
    incoterm_id VARCHAR2(50),
    contract_date DATE,
    effective_date DATE,
    expiry_date DATE,
    total_value NUMBER,
    loading_port VARCHAR2(100),
    destination_port VARCHAR2(100),
    currency VARCHAR2(50),
    total_contract_value NUMBER,
    total_quantity NUMBER,
    status VARCHAR2(50),
    signed_date DATE,
    FOREIGN KEY (customer_id)
    REFERENCES customer(customer_id),
    FOREIGN KEY (payment_term_id)
    REFERENCES payment_term(payment_term_id),
    FOREIGN KEY (staff_id)
    REFERENCES staff(staff_id),
    FOREIGN KEY (contract_type_id)
    REFERENCES contract_type(contract_type_id),
    FOREIGN KEY (incoterm_id)
    REFERENCES incoterm(incoterm_id),
    FOREIGN KEY (proforma_invoice_id)
    REFERENCES proforma_invoice(proforma_invoice_id)
);

CREATE TABLE sale_order (
    sale_order_id VARCHAR2(50) PRIMARY KEY,
    contract_id VARCHAR2(50),
    order_date DATE,
    delivery_date DATE,
    total_amount NUMBER,
    FOREIGN KEY (contract_id)
    REFERENCES contract(contract_id)
);
CREATE TABLE shipment (
    shipment_id VARCHAR2(50) PRIMARY KEY,
    sale_order_id VARCHAR2(50),
    shipment_date DATE,
    carrier VARCHAR2(200),
    tracking_number VARCHAR2(200),
    FOREIGN KEY (sale_order_id)
    REFERENCES sale_order(sale_order_id)
);
CREATE TABLE export_document_set (
    document_set_id VARCHAR2(50) PRIMARY KEY,
    shipment_id VARCHAR2(50),
    issue_date DATE,
    status VARCHAR2(50),
    FOREIGN KEY (shipment_id)
    REFERENCES shipment(shipment_id)
);
CREATE TABLE payment (
    payment_id VARCHAR2(50) PRIMARY KEY,
    payment_term_id VARCHAR2(50),
    sale_order_id VARCHAR2(50),
    payment_date DATE,
    amount NUMBER,
    FOREIGN KEY (payment_term_id)
    REFERENCES payment_term(payment_term_id),
    FOREIGN KEY (sale_order_id)
    REFERENCES sale_order(sale_order_id)
);
