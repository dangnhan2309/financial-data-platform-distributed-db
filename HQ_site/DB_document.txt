-- 1. Loại sản phẩm
CREATE TABLE PRODUCT_TYPE (
    product_type_id VARCHAR2(50) PRIMARY KEY,
    name VARCHAR2(100) NOT NULL,
    description VARCHAR2(500),
    is_active NUMBER(1) DEFAULT 1
) TABLESPACE TS_MASTER;

-- 2. Sản phẩm
CREATE TABLE PRODUCT (
    product_id VARCHAR2(50) PRIMARY KEY,
    product_type_id VARCHAR2(50) REFERENCES PRODUCT_TYPE(product_type_id),
    name VARCHAR2(255) NOT NULL,
    description CLOB,
    price NUMBER(18,2),
    application VARCHAR2(255),
    brix NUMBER(5,2),
    size_val VARCHAR2(50),
    solid NUMBER(5,2),
    ph NUMBER(4,2),
    is_active NUMBER(1) DEFAULT 1,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP
) TABLESPACE TS_MASTER;

-- 3. Quy cách đóng gói
CREATE TABLE PACKAGING_SPECIFICATION (
    packaging_id VARCHAR2(50) PRIMARY KEY,
    name VARCHAR2(100),
    unit_of_measure VARCHAR2(50),
    volume NUMBER(10,2),
    material VARCHAR2(100),
    description VARCHAR2(500),
    is_active NUMBER(1) DEFAULT 1
) TABLESPACE TS_MASTER;

-- 4. Chứng chỉ
CREATE TABLE CERTIFICATION (
    certification_id VARCHAR2(50) PRIMARY KEY,
    name VARCHAR2(255),
    issue_date DATE,
    expiry_date DATE,
    issued_by VARCHAR2(255),
    status VARCHAR2(50),
    description VARCHAR2(500)
) TABLESPACE TS_MASTER;

-- 5. Đảm bảo chất lượng
CREATE TABLE QUALITY_ASSURANCE (
    quality_assurance_id VARCHAR2(50) PRIMARY KEY,
    name VARCHAR2(255),
    description VARCHAR2(500),
    standard_reference VARCHAR2(255),
    status VARCHAR2(50)
) TABLESPACE TS_MASTER;

-- 6. Chi tiết thông số sản phẩm (Bảng liên kết)
CREATE TABLE PRODUCT_SPECIFICATION (
    product_specification_id VARCHAR2(50) PRIMARY KEY,
    product_id VARCHAR2(50) REFERENCES PRODUCT(product_id),
    packaging_id VARCHAR2(50) REFERENCES PACKAGING_SPECIFICATION(packaging_id),
    certification_id VARCHAR2(50) REFERENCES CERTIFICATION(certification_id),
    quality_assurance_id VARCHAR2(50) REFERENCES QUALITY_ASSURANCE(quality_assurance_id),
    shelf_life_days NUMBER,
    storage_condition VARCHAR2(255),
    origin VARCHAR2(100),
    status VARCHAR2(50)
) TABLESPACE TS_MASTER;

-- 7. Loại khách hàng
CREATE TABLE CUSTOMER_TYPE (
    customer_type_id VARCHAR2(50) PRIMARY KEY,
    name VARCHAR2(100),
    description VARCHAR2(500),
    status VARCHAR2(20)
) TABLESPACE TS_MASTER;

-- 8. Khách hàng
CREATE TABLE CUSTOMER (
    customer_id VARCHAR2(50) PRIMARY KEY,
    customer_type_id VARCHAR2(50) REFERENCES CUSTOMER_TYPE(customer_type_id),
    customer_code VARCHAR2(50) UNIQUE,
    company_name VARCHAR2(255) NOT NULL,
    short_name VARCHAR2(100),
    tax_id VARCHAR2(50),
    country VARCHAR2(100),
    city VARCHAR2(100),
    address VARCHAR2(500),
    phone VARCHAR2(20),
    email VARCHAR2(100),
    website VARCHAR2(100),
    industry VARCHAR2(100),
    status VARCHAR2(20),
    preferred_currency VARCHAR2(10),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
) TABLESPACE TS_MASTER;

-- Các bảng danh mục khác
CREATE TABLE STAFF (
    staff_id VARCHAR2(50) PRIMARY KEY,
    name VARCHAR2(255),
    role VARCHAR2(100),
    email VARCHAR2(100),
    phone VARCHAR2(20),
    department VARCHAR2(100),
    status VARCHAR2(20)
) TABLESPACE TS_MASTER;

CREATE TABLE INCOTERM (
    incoterm_id VARCHAR2(50) PRIMARY KEY,
    name VARCHAR2(50),
    description VARCHAR2(500),
    version_val VARCHAR2(20),
    status VARCHAR2(20)
) TABLESPACE TS_MASTER;

CREATE TABLE PAYMENT_TERM (
    payment_term_id VARCHAR2(50) PRIMARY KEY,
    description VARCHAR2(500),
    number_of_days NUMBER,
    status VARCHAR2(20)
) TABLESPACE TS_MASTER;
-- 1. Báo giá
CREATE TABLE QUOTATION (
    quotation_id VARCHAR2(50) PRIMARY KEY,
    customer_id VARCHAR2(50) REFERENCES CUSTOMER(customer_id),
    staff_id VARCHAR2(50) REFERENCES STAFF(staff_id),
    quotation_date DATE,
    expiry_date DATE,
    incoterm_id VARCHAR2(50) REFERENCES INCOTERM(incoterm_id),
    payment_term_id VARCHAR2(50) REFERENCES PAYMENT_TERM(payment_term_id),
    total_amount NUMBER(18,2),
    currency VARCHAR2(10),
    status VARCHAR2(20),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
) TABLESPACE TS_SALES;

-- 2. Chi tiết báo giá
CREATE TABLE QUOTATION_ITEM (
    quotation_id VARCHAR2(50) REFERENCES QUOTATION(quotation_id),
    product_id VARCHAR2(50) REFERENCES PRODUCT(product_id),
    quantity NUMBER(12,2),
    unit_price NUMBER(18,2),
    discount NUMBER(5,2),
    tax_rate NUMBER(5,2),
    total_price NUMBER(18,2),
    PRIMARY KEY (quotation_id, product_id)
) TABLESPACE TS_SALES;

-- 3. Hợp đồng
CREATE TABLE CONTRACT (
    contract_id VARCHAR2(50) PRIMARY KEY,
    customer_id VARCHAR2(50) REFERENCES CUSTOMER(customer_id),
    staff_id VARCHAR2(50) REFERENCES STAFF(staff_id),
    incoterm_id VARCHAR2(50) REFERENCES INCOTERM(incoterm_id),
    payment_term_id VARCHAR2(50) REFERENCES PAYMENT_TERM(payment_term_id),
    contract_date DATE,
    effective_date DATE,
    expiry_date DATE,
    total_contract_value NUMBER(18,2),
    status VARCHAR2(20),
    signed_date DATE
) TABLESPACE TS_SALES;

-- 4. Vận chuyển & Tuyến đường
CREATE TABLE TRANSPORT_ROUTE (
    route_id VARCHAR2(50) PRIMARY KEY,
    route_name VARCHAR2(255),
    origin_location VARCHAR2(255),
    destination_location VARCHAR2(255),
    distance_km NUMBER(10,2),
    status VARCHAR2(20)
) TABLESPACE TS_SALES;

CREATE TABLE SHIPMENT (
    shipment_id VARCHAR2(50) PRIMARY KEY,
    route_id VARCHAR2(50) REFERENCES TRANSPORT_ROUTE(route_id),
    departure_time TIMESTAMP,
    estimated_arrival_time TIMESTAMP,
    status VARCHAR2(20)
) TABLESPACE TS_SALES;

-- 1. Ngân hàng
CREATE TABLE BANK (
    bank_id VARCHAR2(50) PRIMARY KEY,
    bank_name VARCHAR2(255),
    account_name VARCHAR2(255),
    account_number VARCHAR2(50),
    swift_code VARCHAR2(20),
    status VARCHAR2(20)
) TABLESPACE TS_FINANCE;

-- 2. Hóa đơn thương mại
CREATE TABLE COMMERCIAL_INVOICE (
    invoice_id VARCHAR2(50) PRIMARY KEY,
    invoice_date DATE,
    currency VARCHAR2(10),
    total_amount NUMBER(18,2),
    tax_amount NUMBER(18,2),
    due_date DATE,
    status VARCHAR2(20)
) TABLESPACE TS_FINANCE;

-- 3. Thanh toán
CREATE TABLE PAYMENT (
    payment_id VARCHAR2(50) PRIMARY KEY,
    expected_amount NUMBER(18,2),
    paid_amount NUMBER(18,2),
    due_date DATE,
    payment_date DATE,
    status VARCHAR2(20)
) TABLESPACE TS_FINANCE;

-- Bổ sung TS_MASTER
CREATE TABLE COMPANY_TYPE (
    company_type_id VARCHAR2(50) PRIMARY KEY,
    name VARCHAR2(100),
    description VARCHAR2(500),
    status VARCHAR2(20)
) TABLESPACE TS_MASTER;

CREATE TABLE COMPANY (
    company_id VARCHAR2(50) PRIMARY KEY,
    company_type_id VARCHAR2(50) REFERENCES COMPANY_TYPE(company_type_id),
    name VARCHAR2(255),
    short_name VARCHAR2(100),
    tax_id VARCHAR2(50),
    address VARCHAR2(500),
    status VARCHAR2(20)
) TABLESPACE TS_MASTER;

CREATE TABLE FACTORY (
    factory_id VARCHAR2(50) PRIMARY KEY,
    company_id VARCHAR2(50) REFERENCES COMPANY(company_id),
    name VARCHAR2(255),
    address VARCHAR2(500),
    status VARCHAR2(20)
) TABLESPACE TS_MASTER;

CREATE TABLE CONTACT_PERSON (
    contact_person_id VARCHAR2(50) PRIMARY KEY,
    customer_id VARCHAR2(50) REFERENCES CUSTOMER(customer_id),
    name VARCHAR2(255),
    email VARCHAR2(100),
    phone VARCHAR2(20),
    position VARCHAR2(100),
    status VARCHAR2(20)
) TABLESPACE TS_MASTER;

CREATE TABLE CONTRACT_TYPE (
    contract_type_id VARCHAR2(50) PRIMARY KEY,
    name VARCHAR2(100),
    status VARCHAR2(20)
) TABLESPACE TS_MASTER;

-- Bổ sung TS_SALES
CREATE TABLE PROFORMA_INVOICE (
    proforma_invoice_id VARCHAR2(50) PRIMARY KEY,
    quotation_id VARCHAR2(50) REFERENCES QUOTATION(quotation_id),
    bank_id VARCHAR2(50), -- Sẽ ref đến bảng BANK ở TS_FINANCE
    staff_id VARCHAR2(50) REFERENCES STAFF(staff_id),
    total_contract_value NUMBER(18,2),
    currency VARCHAR2(10),
    status VARCHAR2(20),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
) TABLESPACE TS_SALES;

CREATE TABLE CONTRACT_ITEM (
    contract_id VARCHAR2(50) REFERENCES CONTRACT(contract_id),
    product_id VARCHAR2(50) REFERENCES PRODUCT(product_id),
    quantity NUMBER(12,2),
    unit_price NUMBER(18,2),
    tax_rate NUMBER(5,2),
    total_price NUMBER(18,2),
    PRIMARY KEY (contract_id, product_id)
) TABLESPACE TS_SALES;

CREATE TABLE SALE_ORDER (
    sale_order_id VARCHAR2(50) PRIMARY KEY,
    contract_id VARCHAR2(50) REFERENCES CONTRACT(contract_id),
    order_date DATE,
    delivery_date DATE,
    total_amount NUMBER(18,2),
    status VARCHAR2(20)
) TABLESPACE TS_SALES;

-- Bổ sung TS_FINANCE
CREATE TABLE INVOICE_ITEM (
    invoice_item_id VARCHAR2(50) PRIMARY KEY,
    invoice_id VARCHAR2(50) REFERENCES COMMERCIAL_INVOICE(invoice_id),
    product_id VARCHAR2(50) REFERENCES PRODUCT(product_id),
    quantity NUMBER(12,2),
    unit_price NUMBER(18,2),
    total_price NUMBER(18,2)
) TABLESPACE TS_FINANCE;

CREATE TABLE PAYMENT_TRANSACTION (
    transaction_id VARCHAR2(50) PRIMARY KEY,
    payment_id VARCHAR2(50) REFERENCES PAYMENT(payment_id),
    bank_id VARCHAR2(50), -- Sẽ ref đến bảng BANK
    amount NUMBER(18,2),
    payment_date DATE,
    payment_method VARCHAR2(50),
    payment_status VARCHAR2(20),
    reference_number VARCHAR2(100)
) TABLESPACE TS_FINANCE;
